message_setup_this_dir()


# set include directories
include_directories(
	${CMAKE_CURRENT_SOURCE_DIR}
	${RPWA_PARTIALWAVEFIT_INCLUDE_DIR}
	${RPWA_STORAGEFORMATS_INCLUDE_DIR}
	${RPWA_UTILITIES_INCLUDE_DIR}
	SYSTEM
	${Boost_INCLUDE_DIRS}
	${ROOT_INCLUDE_DIR}
	)

add_compile_options(-fopenmp)
add_link_options(-fopenmp)

# source files that are compiled into library
set(SOURCES
	plotcollection.cc
	)


# library name
lib_name_for_current_dir(THIS_LIB)


# ROOT dictionary
set(ROOTPWASTORAGE_DICTIONARY ${CMAKE_CURRENT_BINARY_DIR}/dict.cc)
root_generate_dictionary(
	${ROOTPWASTORAGE_DICTIONARY}
	plotcollection.h
	MODULE ${THIS_LIB}
	LINKDEF linkdef.h
	)
set(SOURCES ${SOURCES} ${ROOTPWASTORAGE_DICTIONARY})


# library
make_shared_library(
	"${THIS_LIB}"
	"${SOURCES}"
	"${ROOT_LIBS}"
	"${RPWA_UTILITIES_LIB}"
	"${RPWA_PARTIALWAVEFIT_LIB}"
	)



make_executable(testplot test.cc ${THIS_LIB} ${RPWA_UTILITIES_LIB})
